// pass the right subclass — the table picks the editor automatically
final columns = <EditableColumn>[
  const ReadonlyColumn(key: 'id', label: 'ID', mono: true),
  const NumericColumn(key: 'qty', label: 'Qty', min: 0, decimals: 0),
  ComputedColumn(key: 'total', label: 'Total', includeInTotal: true,
    compute: (r) => money(qty(r) * price(r))),
  const DateColumn(key: 'due', label: 'Due date'),
  const DropdownColumn(key: 'status', label: 'Status', options: ['Open', 'Done']),
  const ColorPickerColumn(key: 'color', label: 'Colour'),
];

EditableTable(columns: columns, initialRows: seed, showTotals: true,
  confirmDelete: true, growOnTab: true);

Row-aware validation

EditableColumn(key: 'total', label: 'Total',
  cellValidator: (value, row) {
    final q = num(row['qty']);
    final p = num(row['price']);
    return (q * p - num(value)).abs() > 0.01
      ? '≠ Qty × Price' : null;
  });

03 / workspace tabs

A real browser tab strip

Pinned / closable / dirty tabs, drag-to-reorder, a context menu, an overflow dropdown, a dirty-close confirm dialog, and live mini-page previews — the page's real captured frame on hover.

  • Renders only the strip; drives the active body via pageBuilder
  • Any page can drive the strip via BrowserStyleTabBarController.of(context)
  • Pinned tabs · unsaved dots · close-guard dialog · RTL
BrowserStyleTabBar(
  controller: tabs,
  pageBuilder: (ctx, tab) => MyPage(tab),
);
tabs.add(title: 'New report', kind: GLTabKind.chart);
